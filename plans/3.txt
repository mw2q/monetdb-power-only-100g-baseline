	o_orderdate,
	o_shippriority

	customer,
	orders,
	lineitem

	c_mktsegment = 'AUTOMOBILE'
	and c_custkey = o_custkey

	l_orderkey,
	o_orderdate,
	o_shippriority
order by
	revenue desc,
	o_orderdate
