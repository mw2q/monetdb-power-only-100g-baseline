	o_orderpriority,
	count(*) as order_count
from
	orders
where
	o_orderdate >= date '1997-05-01'
	and o_orderdate < date '1997-05-01' + interval '3' month
	and exists (
		select
			*
		from
			lineitem

	)
group by
	o_orderpriority
order by
	o_orderpriority;
